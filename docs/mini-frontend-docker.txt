# Build stage
FROM node:20.20.2-alpine AS build
WORKDIR /app
COPY frontend/package*.json ./
RUN npm ci
COPY frontend/ ./
COPY shared/ ../shared/
ENV VITE_API_URL=http://68.233.103.2:7071/telegram-bot-buddy
ENV VITE_SOCKET_URL=https://playbuddy.zapto.org
ENV VITE_LOGGER_URL=http://68.233.103.2:7071/telegram-bot-buddy/logs/print
RUN npm run build

# Production stage
FROM nginx:alpine
COPY --from=build /app/dist /usr/share/nginx/html
COPY docker/nginx.conf /etc/nginx/conf.d/default.conf
EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]
